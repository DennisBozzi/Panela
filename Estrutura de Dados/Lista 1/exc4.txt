using System;

class Program
{
    public static void Main(string[] args)
    {

        int[] idades = new int[7];

        int maior = 0;

        Console.WriteLine("Digite 7 idades:");
        Console.WriteLine();

        for (int i = 0; i < idades.Length; i++)
        {
            idades[i] = Convert.ToInt32(Console.ReadLine());

            if (idades[i] > maior)
            {
                maior = idades[i];
            }
        }

        Console.WriteLine();
        Console.WriteLine("A maior idade Ã©: " + maior);

    }
}
